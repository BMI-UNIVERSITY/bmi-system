      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4

      - run: npm ci
      - run: npm run lint
      - run: npm test
      - run: npm run build

  docker:
    name: Docker build and non-root verification
    runs-on: ubuntu-latest

          docker compose -f docker-compose.yml down
